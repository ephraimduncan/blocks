            </div>

              <CommandPrimitive.Empty className="py-6 text-center text-sm text-muted-foreground">
                No results found.
              </CommandPrimitive.Empty>

              {/* Top level items (no group heading) */}
              <CommandItem value="settings" onSelect={() => setOpen(false)}>
                <ItemIcon>
                  <IconSettings className="size-4" />
                </ItemIcon>
                Account Settings...
              </CommandItem>
              <CommandItem
                value="switch workspace"
                onSelect={() => setOpen(false)}
              >

                  <IconUser className="size-4" />
                </ItemIcon>
                Switch Workspace...
              </CommandItem>
              <CommandItem value="log out" onSelect={() => setOpen(false)}>
                <ItemIcon>
                  <IconLogout className="size-4" />

                Log Out
              </CommandItem>

              {/* Documents Group */}

                >
                  <ItemIcon>

                  </ItemIcon>
                  Search Documents...
                  <Keybinds>
                    <Kbd>⇧</Kbd>
                    <Kbd>D</Kbd>
                  </Keybinds>
                </CommandItem>

                  onSelect={() => setOpen(false)}

                  <ItemIcon>

                  Create New Document...

                >

                  </ItemIcon>

                </CommandItem>
              </CommandGroup>

              {/* Signing Group */}

                  </ItemIcon>

                </CommandItem>
              </CommandGroup>

              {/* Templates Group */}

// Helper components for exact styling match
function CommandGroup({
  heading,
  children,
}: {
  heading: string;
  children: React.ReactNode;
}) {
  return (
    <CommandPrimitive.Group className="px-2 py-1.5">
      <div className="px-2 py-1.5 text-xs font-medium text-muted-foreground">
        {heading}
      </div>
      {children}
    </CommandPrimitive.Group>
  );
}

function CommandItem({
  value,
  children,
  onSelect,
}: {
  value: string;
  children: React.ReactNode;
  onSelect?: () => void;
}) {
  return (
    <CommandPrimitive.Item
      value={value}
      onSelect={onSelect}
      className="relative flex cursor-pointer items-center gap-2 rounded-lg px-2 py-2.5 text-sm text-foreground outline-none select-none data-[selected=true]:bg-accent mx-2"
    >
      {children}
    </CommandPrimitive.Item>
  );
}

function ItemIcon({ children }: { children: React.ReactNode }) {
  return (
    <div className="flex items-center justify-center w-4 h-4 text-muted-foreground">
      {children}
    </div>
  );
}

function Keybinds({ children }: { children: React.ReactNode }) {
  return <div className="ml-auto flex items-center gap-1">{children}</div>;
}

function Kbd({ children }: { children: React.ReactNode }) {
  return (
    <kbd className="h-5 min-w-5 flex items-center justify-center rounded bg-muted px-1.5 text-[11px] font-medium text-muted-foreground border border-border/50">
      {children}
    </kbd>
  );
}
